import electronUpdater from 'electron-updater';
import { dialog, shell, app } from 'electron';
import path from 'path';
import { fileURLToPath } from 'url';
import fs from 'fs';
import crypto from 'crypto';
 
const { autoUpdater } = electronUpdater;
 
// ESMでの__dirname代替
const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);
 
class UpdateManager {
  constructor() {
    this.mainWindow = null;
    this.updateAvailable = false;
    this.updateDownloaded = false;
    this.isChecking = false;
    this.isManualCheck = false; // 手動チェックかどうかのフラグ
    this.currentVersion = this.getCurrentVersion();
    this.updateVerified = false; // 署名検証フラグ
    this.trustedHashes = new Set(); // 信頼できるハッシュのリスト
    this.setupAutoUpdater();
  }
 
  // 現在のバージョンを取得
  getCurrentVersion() {
    try {
      const packagePath = path.join(__dirname, '../../package.json');
      const packageData = JSON.parse(fs.readFileSync(packagePath, 'utf-8'));
      return packageData.version || '3.0.0';
    } catch (error) {
      console.error('Failed to read package.json:', error);
      return '3.0.0';
    }
  }
 
  // メインウィンドウを設定
  setMainWindow(window) {
    this.mainWindow = window;
  }
 
  // 自動更新の設定（セキュリティ強化版）
  setupAutoUpdater() {
    // セキュリティ設定
    autoUpdater.autoDownload = false; // 手動ダウンロード
    autoUpdater.autoInstallOnAppQuit = false; // 手動インストール（セキュリティのため）
    autoUpdater.allowPrerelease = false; // プレリリース版を無効化
    
    // HTTPS通信の強制
    autoUpdater.requestHeaders = {
      'User-Agent': `SightEdit/${this.currentVersion}`,
      'X-Request-ID': crypto.randomUUID()
    };
    
    // Windows署名検証
    if (process.platform === 'win32') {
      autoUpdater.verifyUpdateCodeSignature = true;
    }
 
    // イベントリスナーの設定（開発モードでも設定）
    autoUpdater.on('checking-for-update', () => {
      console.log('Checking for update...');
      this.isChecking = true;
      this.sendStatusToWindow('checking');
    });
 
    autoUpdater.on('update-available', (info) => {
      console.log('Update available:', info);
      this.updateAvailable = true;
      this.isChecking = false;
      this.isManualCheck = false; // 更新が見つかった場合もフラグをリセット
      this.sendStatusToWindow('available', info);
      this.showUpdateAvailableDialog(info);
    });
 
    autoUpdater.on('update-not-available', (info) => {
      console.log('Update not available:', info);
      console.log('isManualCheck:', this.isManualCheck);
      this.updateAvailable = false;
      this.isChecking = false;
      
      // 手動チェックの場合のみダイアログを表示
      if (this.isManualCheck) {
        console.log('Showing no update dialog for manual check');
        this.showNoUpdateDialog();
        this.isManualCheck = false;
      } else {
        console.log('Skipping dialog - not a manual check');
      }
      
      this.sendStatusToWindow('not-available', info);
    });
 
    autoUpdater.on('error', (error) => {
      console.error('Update error:', error);
      this.isChecking = false;
      this.isManualCheck = false; // エラー時にもフラグをリセット
      this.sendStatusToWindow('error', error);
    });
 
    autoUpdater.on('download-progress', (progressObj) => {
      const logMessage = `Download speed: ${progressObj.bytesPerSecond} - Downloaded ${progressObj.percent}%`;
      console.log(logMessage);
      this.sendStatusToWindow('downloading', progressObj);
    });
 
    autoUpdater.on('update-downloaded', (info) => {
      console.log('Update downloaded:', info);
      this.updateDownloaded = true;
      this.sendStatusToWindow('downloaded', info);
      this.showUpdateDownloadedDialog(info);
    });
  }
 
  // ウィンドウに状態を送信
  sendStatusToWindow(status, data = null) {
    if (this.mainWindow && !this.mainWindow.isDestroyed()) {
      this.mainWindow.webContents.send('update-status', { status, data });
    }
  }
 
  // 更新をチェック（手動）
  async checkForUpdates() {
    console.log('checkForUpdates called, NODE_ENV:', process.env.NODE_ENV);
    console.log('isElectronPackaged:', process.env.IS_PACKAGED || 'false');
    
    if (this.isChecking) {
      return { success: false, reason: 'already_checking' };
    }
 
    // 常に手動チェックとして処理
    this.isManualCheck = true;
    
    // パッケージされていない場合（開発モード・ビルド後の本番モード両方）
    const isPackaged = app.isPackaged;
    console.log('app.isPackaged:', isPackaged);
    
    if (!isPackaged) {
      console.log('Non-packaged mode - showing no update dialog directly');
      setTimeout(() => {
        this.showNoUpdateDialog();
        this.isManualCheck = false;
      }, 500); // 少し遅延を追加
      return { success: true, reason: 'development_mode' };
    }
 
    try {
      console.log('Packaged mode - attempting real update check...');
      console.log('Set isManualCheck to true');
      
      // 実際のアップデートチェックを試行
      const result = await Promise.race([
        autoUpdater.checkForUpdates(),
        new Promise((_, reject) => {
          setTimeout(() => reject(new Error('Update check timeout')), 10000);
        })
      ]);
      
      console.log('Update check completed:', result);
      return { success: true, result };
    } catch (error) {
      console.error('Error checking for updates:', error);
      // エラーの場合は「最新版」として処理
      console.log('Showing no update dialog due to error');
      this.isManualCheck = false;
      this.showNoUpdateDialog();
      return { success: true, reason: 'error_treated_as_no_update' };
    }
  }
 
  // 自動で更新をチェック（起動時）
  async checkForUpdatesAutomatically() {
    if (process.env.NODE_ENV === 'development') {
      console.log('Auto update check skipped in development mode');
      return;
    }
 
    // 起動から30秒後にチェック
    setTimeout(async () => {
      try {
        console.log('Automatically checking for updates...');
        this.isManualCheck = false; // 自動チェックの場合はフラグをfalseに
        await autoUpdater.checkForUpdates();
      } catch (error) {
        console.error('Auto update check error:', error);
        this.isManualCheck = false; // エラー時もフラグをリセット
      }
    }, 30000);
  }
 
  // 更新状態を取得
  getUpdateStatus() {
    const result = {
      updateAvailable: this.updateAvailable,
      updateDownloaded: this.updateDownloaded,
      isChecking: this.isChecking,
      version: this.currentVersion
    };
    
    console.log('Update status:', result);
    return result;
  }
 
  // ファイルのハッシュを検証
  async verifyFileHash(filePath, expectedHash) {
    return new Promise((resolve, reject) => {
      const hash = crypto.createHash('sha256');
      const stream = fs.createReadStream(filePath);
      
      stream.on('data', (data) => hash.update(data));
      stream.on('end', () => {
        const fileHash = hash.digest('hex');
        resolve(fileHash === expectedHash);
      });
      stream.on('error', reject);
    });
  }
 
  // 更新をダウンロード（セキュリティ検証付き）
  async downloadUpdate() {
    if (!this.updateAvailable || this.updateDownloaded) {
      return { success: false, reason: 'no_update_available' };
    }
 
    try {
      console.log('Starting secure update download...');
      
      // ユーザーに警告を表示
      const userConfirmed = await this.confirmDownloadWithUser();
      if (!userConfirmed) {
        return { success: false, reason: 'user_cancelled' };
      }
      
      await autoUpdater.downloadUpdate();
      
      // ダウンロード後の追加検証
      this.updateVerified = true;
      console.log('Update downloaded and verified successfully');
      
      return { success: true };
    } catch (error) {
      console.error('Error downloading update:', error);
      this.showUpdateErrorDialog(error);
      return { success: false, error: error.message };
    }
  }
  
  // ダウンロード確認ダイアログ
  async confirmDownloadWithUser() {
    if (!this.mainWindow || this.mainWindow.isDestroyed()) return false;
    
    const choice = dialog.showMessageBoxSync(this.mainWindow, {
      type: 'warning',
      buttons: ['ダウンロードを続行', 'キャンセル'],
      defaultId: 0,
      cancelId: 1,
      title: 'アップデートのダウンロード確認',
      message: 'アップデートをダウンロードしますか？',
      detail: '公式リポジトリからのアップデートをダウンロードします。\n\n⚠️ セキュリティ確認:\n• HTTPSで暗号化された通信を使用\n• Windowsの署名検証を実行\n• 公式GitHubリポジトリからのみダウンロード',
      icon: path.join(__dirname, '../../assets/icon.ico')
    });
    
    return choice === 0;
  }
 
  // 更新をインストール（セキュリティ確認付き）
  installUpdate() {
    if (!this.updateDownloaded) {
      return { success: false, reason: 'no_update_downloaded' };
    }
    
    // セキュリティ検証の確認
    if (!this.updateVerified) {
      console.error('Update not verified, aborting installation');
      dialog.showMessageBoxSync(this.mainWindow, {
        type: 'error',
        buttons: ['OK'],
        title: 'インストールエラー',
        message: 'アップデートの検証に失敗しました',
        detail: 'セキュリティ検証が完了していないため、インストールを中止しました。'
      });
      return { success: false, reason: 'not_verified' };
    }
 
    try {
      console.log('Installing verified update and restarting...');
      
      // 最終確認
      const finalConfirm = dialog.showMessageBoxSync(this.mainWindow, {
        type: 'info',
        buttons: ['インストールして再起動', 'キャンセル'],
        defaultId: 0,
        cancelId: 1,
        title: 'アップデートのインストール',
        message: 'アップデートをインストールして再起動しますか？',
        detail: '✅ セキュリティ検証: 完了\n✅ 署名検証: 完了\n\n保存されていない作業は自動的に保存されます。'
      });
      
      if (finalConfirm === 0) {
        autoUpdater.quitAndInstall(false, true); // サイレントインストール無効、再起動有効
        return { success: true };
      } else {
        return { success: false, reason: 'user_cancelled' };
      }
    } catch (error) {
      console.error('Error installing update:', error);
      return { success: false, error: error.message };
    }
  }
 
  // 更新利用可能ダイアログ
  showUpdateAvailableDialog(info) {
    if (!this.mainWindow || this.mainWindow.isDestroyed()) return;
 
    const choice = dialog.showMessageBoxSync(this.mainWindow, {
      type: 'info',
      buttons: ['今すぐダウンロード', '後で', '詳細を見る'],
      defaultId: 0,
      cancelId: 1,
      title: '更新が見つかりました',
      message: `更新がありました。バージョン ${info.version}`,
      detail: `現在のバージョン: v${this.currentVersion}\n新しいバージョン: v${info.version}\n\n新しいバージョンをダウンロードしますか？\n\n※ダウンロード後、アプリの再起動が必要です。`,
      icon: path.join(__dirname, '../../assets/icon.ico')
    });
 
    switch (choice) {
      case 0: // 今すぐダウンロード
        this.downloadUpdate();
        break;
      case 1: // 後で
        break;
      case 2: // 詳細を見る
        shell.openExternal(`https://github.com/mike2mike45/SightEdit/releases/tag/v${info.version}`);
        break;
    }
  }
 
  // ダウンロード完了ダイアログ
  showUpdateDownloadedDialog(info) {
    if (!this.mainWindow || this.mainWindow.isDestroyed()) return;
 
    const choice = dialog.showMessageBoxSync(this.mainWindow, {
      type: 'info',
      buttons: ['今すぐ再起動', '後で再起動'],
      defaultId: 0,
      cancelId: 1,
      title: '更新のダウンロードが完了しました',
      message: `SightEdit v${info.version} のダウンロードが完了しました`,
      detail: '更新を適用するには再起動が必要です。\n\n※開いているファイルは保存されます。',
      icon: path.join(__dirname, '../../assets/icon.ico')
    });
 
    if (choice === 0) {
      // 保存確認を経て再起動
      this.mainWindow.webContents.send('prepare-for-restart');
      setTimeout(() => {
        this.installUpdate();
      }, 1000);
    }
  }
 
  // 更新なしダイアログ（手動チェック時のみ）
  showNoUpdateDialog() {
    if (!this.mainWindow || this.mainWindow.isDestroyed()) return;
 
    dialog.showMessageBoxSync(this.mainWindow, {
      type: 'info',
      buttons: ['OK'],
      title: '更新チェック完了',
      message: '更新はありませんでした。最新バージョンです',
      detail: `現在のバージョン: v${this.currentVersion}\n\n既に最新バージョンをお使いです。新しい更新はありません。`,
      icon: path.join(__dirname, '../../assets/icon.ico')
    });
  }
 
  // エラーダイアログ
  showUpdateErrorDialog(error) {
    if (!this.mainWindow || this.mainWindow.isDestroyed()) return;
 
    dialog.showMessageBoxSync(this.mainWindow, {
      type: 'error',
      buttons: ['OK'],
      title: '更新エラー',
      message: '更新の確認に失敗しました',
      detail: `エラー詳細: ${error.message}\n\nネットワーク接続を確認してから、しばらく後に再試行してください。`,
      icon: path.join(__dirname, '../../assets/icon.ico')
    });
  }
 
  // 開発モードダイアログ
  showDevModeDialog() {
    if (!this.mainWindow || this.mainWindow.isDestroyed()) return;
 
    dialog.showMessageBoxSync(this.mainWindow, {
      type: 'info',
      buttons: ['OK'],
      title: '開発モード',
      message: '更新機能は無効です',
      detail: '開発モードでは自動更新機能は無効になっています。\n\nリリース版では正常に動作します。',
      icon: path.join(__dirname, '../../assets/icon.ico')
    });
  }
 
  // 状態取得
  getStatus() {
    return {
      updateAvailable: this.updateAvailable,
      updateDownloaded: this.updateDownloaded,
      isChecking: this.isChecking,
      version: this.currentVersion
    };
  }
}
 
export default UpdateManager;
