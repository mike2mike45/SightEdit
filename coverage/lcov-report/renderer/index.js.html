    'bold': /\*\*[^*]+\*\*/gms,           // 
    'italic': /(?<!\*)\*[^*\n]+\*(?!\*)/gm,  // 
    'strike': /~~[^~]+~~/gms,              // 
    'inline-code': /`[^`]+`/gm             // 
  };
  
  // 
  for (const [format, regex] of Object.entries(formatChecks)) {
    if (format !== actionType && regex.test(text)) {
      return true; // 
    }
  }
  
  return false;
}

function applySourceFormatting(actionType, params = {}) {
  if (!sourceEditorEl) {
    console.warn('Source editor element not available for formatting');
    return;
  }
  
  const text = sourceEditorEl.value;
  const start = sourceEditorEl.selectionStart;
  const end = sourceEditorEl.selectionEnd;
  const selectedText = text.substring(start, end);
  
  // 
  const inlineFormats = ['bold', 'italic', 'strike', 'inline-code'];
  if (inlineFormats.includes(actionType)) {
    // 
    if (!selectedText) {
      if (window.showMessage) {
        window.showMessage('', 'warning');
      }
      return;
    }
    
    // 
    if (selectedText.includes('\n')) {
      // 
      if (actionType === 'italic' || actionType === 'inline-code') {
        if (window.showMessage) {
          window.showMessage(`${actionType === 'italic' ? '' : ''}`, 'warning');
        }
        return;
      }
      // 
      const lines = selectedText.split('\n');
      for (const line of lines) {
        if (line && hasConflictingFormat(line, actionType)) {
          if (window.showMessage) {
            window.showMessage('', 'warning');
          }
          return;
        }
      }
    }
  }
  
  // 
  if (selectedText && actionType !== 'blockquote' && !selectedText.includes('\n')) {
    if (hasConflictingFormat(selectedText, actionType)) {
      if (window.showMessage) {
        window.showMessage('', 'warning');
      }
      return;
    }
  }
  
  let newText = text;
  let newStart = start;
  let newEnd = end;
  
  switch(actionType) {
    case 'bold':
      if (selectedText) {
        // **
        if (selectedText.startsWith('**') && selectedText.endsWith('**') && selectedText.length > 4) {
          const innerText = selectedText.slice(2, -2);
          // 
          if (hasConflictingFormat(innerText, 'bold')) {
            if (window.showMessage) {
              window.showMessage('', 'warning');
            }
            return;
          }
          // 
          newText = text.substring(0, start) + innerText + text.substring(end);
          newEnd = start + innerText.length;
        } else {
          // 
          newText = text.substring(0, start) + '**' + selectedText + '**' + text.substring(end);
          newEnd = end + 4;
        }
      }
      break;
      
    case 'italic':
      if (selectedText) {
        // *
        if (selectedText.startsWith('*') && selectedText.endsWith('*') && selectedText.length > 2 && !selectedText.startsWith('**')) {
          const innerText = selectedText.slice(1, -1);
          // 
          if (hasConflictingFormat(innerText, 'italic')) {
            if (window.showMessage) {
              window.showMessage('', 'warning');
            }
            return;
          }
          // 
          newText = text.substring(0, start) + innerText + text.substring(end);
          newEnd = start + innerText.length;
        } else {
          // 
          newText = text.substring(0, start) + '*' + selectedText + '*' + text.substring(end);
          newEnd = end + 2;
        }
      }
      break;
      
    case 'strike':
      if (selectedText) {
        // ~~
        if (selectedText.startsWith('~~') && selectedText.endsWith('~~') && selectedText.length > 4) {
          const innerText = selectedText.slice(2, -2);
          // 
          if (hasConflictingFormat(innerText, 'strike')) {
            if (window.showMessage) {
              window.showMessage('', 'warning');
            }
            return;
          }
          // 
          newText = text.substring(0, start) + innerText + text.substring(end);
          newEnd = start + innerText.length;
        } else {
          // 
          newText = text.substring(0, start) + '~~' + selectedText + '~~' + text.substring(end);
          newEnd = end + 4;
        }
      }
      break;
      
    case 'inline-code':
      if (selectedText) {
        // `
        if (selectedText.startsWith('`') && selectedText.endsWith('`') && selectedText.length > 2) {
          // 
          const innerText = selectedText.slice(1, -1);
          newText = text.substring(0, start) + innerText + text.substring(end);
          newEnd = start + innerText.length;
        } else {
          // 
          newText = text.substring(0, start) + '`' + selectedText + '`' + text.substring(end);
          newEnd = end + 2;
        }
      }
      break;
      
    case 'heading':
      const level = params.level || 1;
      const prefix = '#'.repeat(level) + ' ';
      // 
      let lineStart = text.lastIndexOf('\n', start - 1) + 1;
      let lineEnd = text.indexOf('\n', end);
      if (lineEnd === -1) lineEnd = text.length;
      
      const currentLine = text.substring(lineStart, lineEnd);
      // 
      const cleanLine = currentLine.replace(/^#+\s*/, '');
      
      // 
      if (currentLine.startsWith(prefix)) {
        newText = text.substring(0, lineStart) + cleanLine + text.substring(lineEnd);
        newStart = lineStart;
        newEnd = lineStart + cleanLine.length;
      } else {
        newText = text.substring(0, lineStart) + prefix + cleanLine + text.substring(lineEnd);
        newStart = lineStart + prefix.length;
        newEnd = lineStart + prefix.length + cleanLine.length;
      }
      break;
      
    case 'bullet-list':
      // 
      let blStart = text.lastIndexOf('\n', start - 1) + 1;
      let blEnd = text.indexOf('\n', end);
      if (blEnd === -1) blEnd = text.length;
      
      const blLine = text.substring(blStart, blEnd);
      const blClean = blLine.replace(/^[-*+]\s*/, '').replace(/^\d+\.\s*/, '');
      
      // 
      if (blLine.match(/^[-*+]\s/)) {
        newText = text.substring(0, blStart) + blClean + text.substring(blEnd);
        newStart = blStart;
        newEnd = blStart + blClean.length;
      } else {
        newText = text.substring(0, blStart) + '- ' + blClean + text.substring(blEnd);
        newStart = blStart + 2;
        newEnd = blStart + 2 + blClean.length;
      }
      break;
      
    case 'ordered-list':
      // 
      let olStart = text.lastIndexOf('\n', start - 1) + 1;
      let olEnd = text.indexOf('\n', end);
      if (olEnd === -1) olEnd = text.length;
      
      const olLine = text.substring(olStart, olEnd);
      const olClean = olLine.replace(/^[-*+]\s*/, '').replace(/^\d+\.\s*/, '');
      
      // 
      if (olLine.match(/^\d+\.\s/)) {
        newText = text.substring(0, olStart) + olClean + text.substring(olEnd);
        newStart = olStart;
        newEnd = olStart + olClean.length;
      } else {
        newText = text.substring(0, olStart) + '1. ' + olClean + text.substring(olEnd);
        newStart = olStart + 3;
        newEnd = olStart + 3 + olClean.length;
      }
      break;
      
    case 'task-list':
      // 
      let tlStart = text.lastIndexOf('\n', start - 1) + 1;
      let tlEnd = text.indexOf('\n', end);
      if (tlEnd === -1) tlEnd = text.length;
      
      const tlLine = text.substring(tlStart, tlEnd);
      const tlClean = tlLine.replace(/^[-*+]\s*(\[[x ]\]\s*)?/, '').replace(/^\d+\.\s*/, '');
      
      // 
      if (tlLine.match(/^[-*+]\s*\[[x ]\]/)) {
        newText = text.substring(0, tlStart) + tlClean + text.substring(tlEnd);
        newStart = tlStart;
        newEnd = tlStart + tlClean.length;
      } else {
        newText = text.substring(0, tlStart) + '- [ ] ' + tlClean + text.substring(tlEnd);
        newStart = tlStart + 6;
        newEnd = tlStart + 6 + tlClean.length;
      }
      break;
      
    case 'code-block':
      if (selectedText) {
        // 
        const beforeText = text.substring(Math.max(0, start - 10), start);
        const afterText = text.substring(end, Math.min(text.length, end + 10));
        if (beforeText.includes('```') && afterText.includes('```')) {
          // 
          newText = text.substring(0, start) + selectedText + text.substring(end);
        } else {
          newText = text.substring(0, start) + '\n```\n' + selectedText + '\n```\n' + text.substring(end);
          newEnd = end + 10;
        }
      } else {
        newText = text.substring(0, start) + '\n```\n\n```\n' + text.substring(end);
        newStart = start + 5;
        newEnd = start + 5;
      }
      break;
      
    case 'blockquote':
      // 
      let bqStart = text.lastIndexOf('\n', start - 1) + 1;
      let bqEnd = text.indexOf('\n', end);
      if (bqEnd === -1) bqEnd = text.length;
      
      const bqLine = text.substring(bqStart, bqEnd);
      
      // 
      const quoteMatch = bqLine.match(/^(>\s*)+/);
      const currentQuoteLevel = quoteMatch ? quoteMatch[0].length / 2 : 0;
      
      if (currentQuoteLevel > 0) {
        // 
        if (currentQuoteLevel >= 3) {
          // 3
          const bqClean = bqLine.replace(/^>\s*/, '');
          newText = text.substring(0, bqStart) + bqClean + text.substring(bqEnd);
          newStart = bqStart;
          newEnd = bqStart + bqClean.length;
        } else {
          // 3
          newText = text.substring(0, bqStart) + '> ' + bqLine + text.substring(bqEnd);
          newStart = bqStart + 2;
          newEnd = bqStart + 2 + bqLine.length;
        }
      } else {
        // 
        newText = text.substring(0, bqStart) + '> ' + bqLine + text.substring(bqEnd);
        newStart = bqStart + 2;
        newEnd = bqStart + 2 + bqLine.length;
      }
      break;
      
    case 'horizontal-rule':
      // 
      const hrText = '\n\n---\n\n';
      newText = text.substring(0, start) + hrText + text.substring(end);
      newStart = start + hrText.length;
      newEnd = newStart;
      break;
      
    case 'undo':
      // undo
      if (sourceEditorEl && typeof sourceEditorEl.undo === 'function') {
        sourceEditorEl.undo();
      } else {
        // : Ctrl+Z 
        const undoEvent = new KeyboardEvent('keydown', {
          key: 'z',
          ctrlKey: true,
          bubbles: true
        });
        sourceEditorEl?.dispatchEvent(undoEvent);
      }
      return;
      
    case 'redo':
      // redo
      if (sourceEditorEl && typeof sourceEditorEl.redo === 'function') {
        sourceEditorEl.redo();
      } else {
        // : Ctrl+Y 
        const redoEvent = new KeyboardEvent('keydown', {
          key: 'y',
          ctrlKey: true,
          bubbles: true
        });
        sourceEditorEl?.dispatchEvent(redoEvent);
      }
      return;
    
    default:
      console.warn(`Unknown formatting action: ${actionType}`);
      return;
  }
  
  // 
  sourceEditorEl.value = newText;
  sourceEditorEl.setSelectionRange(newStart, newEnd);
  
  // 
  const event = new Event('input', { bubbles: true });
  sourceEditorEl.dispatchEvent(event);
  
  // 
  updateStatus(false);
}

function setupAIDropdownMenu() {
  const aiMenuBtn = document.getElementById('ai-menu-btn');
  const aiDropdownMenu = document.getElementById('ai-dropdown-menu');
  
  if (!aiMenuBtn || !aiDropdownMenu) {
    console.warn('AI menu elements not found');
    return;
  }
 
  // 
  aiMenuBtn.addEventListener('click', (e) => {
    e.stopPropagation();
    const isVisible = aiDropdownMenu.style.display === 'block';
    aiDropdownMenu.style.display = isVisible ? 'none' : 'block';
  });
 
  // AI
  const aiMenuItems = aiDropdownMenu.querySelectorAll('.ai-menu-item');
  aiMenuItems.forEach(item => {
    item.addEventListener('click', (e) => {
      e.stopPropagation();
      
      // 
      if (item.classList.contains('has-submenu')) {
        return;
      }
      
      const aiFunction = item.getAttribute('data-ai-function');
      const option = item.getAttribute('data-option');
      
      // 
      aiDropdownMenu.style.display = 'none';
      // 
      const allSubmenus = aiDropdownMenu.querySelectorAll('.ai-submenu');
      allSubmenus.forEach(submenu => submenu.style.display = 'none');
      
      // AI
      if (aiFunction === 'custom') {
        // 
        if (aiCustomDialog) {
          aiCustomDialog.show();
        }
      } else if (aiFunction === 'settings') {
        // AI
        if (typeof window.showSetupDialog === 'function') {
          window.showSetupDialog();
        } else {
          console.error('AI setup dialog not available');
          showMessage('AI', 'error');
        }
      } else {
        // AI
        executeAIFunction(aiFunction, option);
      }
    });
  });
 
  // 
  const submenuItems = aiDropdownMenu.querySelectorAll('.ai-menu-item-with-submenu');
  submenuItems.forEach(submenuItem => {
    const submenu = submenuItem.querySelector('.ai-submenu');
    if (submenu) {
      submenuItem.addEventListener('mouseenter', () => {
        // 
        const allSubmenus = aiDropdownMenu.querySelectorAll('.ai-submenu');
        allSubmenus.forEach(sm => {
          if (sm !== submenu) sm.style.display = 'none';
        });
        // 
        submenu.style.display = 'block';
      });
 
      submenuItem.addEventListener('mouseleave', (e) => {
        // 
        setTimeout(() => {
          if (!submenuItem.matches(':hover')) {
            submenu.style.display = 'none';
          }
        }, 100);
      });
    }
  });
 
  // 
  document.addEventListener('click', (e) => {
    if (!aiMenuBtn.contains(e.target) && !aiDropdownMenu.contains(e.target)) {
      aiDropdownMenu.style.display = 'none';
    }
  });
  
  // ESC
  document.addEventListener('keydown', (e) => {
    if (e.key === 'Escape' && aiDropdownMenu.style.display === 'block') {
      aiDropdownMenu.style.display = 'none';
    }
  });
}
